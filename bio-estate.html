/* ─── CORE SCANNER LOGIC & TELEMETRY ─── */ // Pointing directly to your primary domain gateway on the verified working endpoint const API_URL = "https://www.francefarms.net/api/telemetry"; const dropZone = document.getElementById('drop-zone'); const fileInput = document.getElementById('file-input'); const terminalConsole = document.getElementById('terminal-console'); const previewImg = document.getElementById('preview-img'); const lensGlyph = document.getElementById('lens-glyph'); const executeBtn = document.getElementById('execute-btn'); // Custom Logger targeting your UI terminal component function consoleLog(msg, type = '') { const entry = document.createElement('div'); entry.className = `log-entry ${type}`; entry.textContent = msg; terminalConsole.appendChild(entry); terminalConsole.scrollTop = terminalConsole.scrollHeight; } // File Upload Visual Triggers dropZone.addEventListener('click', () => fileInput.click()); fileInput.addEventListener('change', (e) => { const file = e.target.files[0]; if (file) { const reader = new FileReader(); reader.onload = (event) => { previewImg.src = event.target.result; previewImg.style.display = 'block'; lensGlyph.style.display = 'none'; document.getElementById('drop-text').style.display = 'none'; consoleLog(`> Image bound successfully: ${file.name}`); }; reader.readAsDataURL(file); } }); // GPS Telemetry Execution Trigger executeBtn.addEventListener('click', () => { consoleLog("> Initializing Geolocation verification protocol..."); if (!navigator.geolocation) { consoleLog("[ERROR] Geolocation is blocked or unsupported by this browser environment.", "log-error"); return; } navigator.geolocation.getCurrentPosition( async (position) => { const latitude = position.coords.latitude; const longitude = position.coords.longitude; const accuracy = position.coords.accuracy; consoleLog(`[SUCCESS] GPS Fixed: Lat ${latitude.toFixed(4)}°, Lon ${longitude.toFixed(4)}° (Accuracy: ±${accuracy.toFixed(1)}m)`, "log-success"); // Proceed to asynchronously construct and transmit your sealed data payload await sendDataToBackend(latitude, longitude); }, (error) => { switch(error.code) { case error.PERMISSION_DENIED: consoleLog("[ERROR] Access Denied: Operator rejected location permissions.", "log-error"); alert("Françe Farms requires device location access to authenticate batch provenance."); break; case error.POSITION_UNAVAILABLE: consoleLog("[ERROR] Telemetry Failed: Satellite linkage unavailable.", "log-error"); break; case error.TIMEOUT: consoleLog("[ERROR] Telemetry Failed: Core location handshake timed out.", "log-error"); break; } }, { enableHighAccuracy: true, timeout: 12000, maximumAge: 0 } ); }); // Helper Function: Generates a secure cryptographic SHA-256 string inside browsers async function generateSHA256(message) { const msgBuffer = new TextEncoder().encode(message); const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer); const hashArray = Array.from(new Uint8Array(hashBuffer)); return hashArray.map(b => b.toString(16).padStart(2, '0')).join(''); } async function sendDataToBackend(lat, lon) { const nodeId = document.getElementById('node-id').value || "UNASSIGNED_NODE"; const batchRef = document.getElementById('batch').value || "GENERIC_BATCH"; const isoTimestamp = new Date().toISOString(); // 1. Create a unique raw data string to seed the hash engine const rawStringSeed = `${nodeId}-${batchRef}-${lat}-${lon}-${isoTimestamp}`; consoleLog(`> Running local cryptographic hashing protocol...`); // 2. Compute the immutable SHA-256 hash string const clientGeneratedHash = await generateSHA256(rawStringSeed); // 3. Compile the secure, unalterable ledger payload block const payload = { node_id: nodeId, timestamp: isoTimestamp, event_type: `SCAN_VERIFICATION:${batchRef}`, payload_hash: clientGeneratedHash }; consoleLog(`[CRYPTO] Secure Hash Sealed: ${clientGeneratedHash}`, "log-success"); consoleLog(`> Payload block compiled cleanly. Transmitting telemetry profiles to database backend...`); try { // 4. Live fetch transmission engine pipeline const res = await fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); if (res.ok) { const data = await res.json().catch(() => ({})); consoleLog(`[DATABASE] Record committed successfully!`, "log-success"); } else { const errData = await res.text(); consoleLog(`[SERVER ERROR] Status ${res.status}: ${errData}`, "log-error"); } } catch (err) { consoleLog(`[NETWORK ERROR] Could not establish connection to API Gateway interface.`, "log-error"); console.error(err); } }
